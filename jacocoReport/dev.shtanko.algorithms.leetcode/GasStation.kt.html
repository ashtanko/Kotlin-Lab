package dev.shtanko.algorithms.leetcode

/**
 * 134. Gas Station
 * @see <a href="https://leetcode.com/problems/gas-station/">leetcode page</a>
 */
fun interface GasStation {
    fun canCompleteCircuit(gas: IntArray, cost: IntArray): Int
}

class GasStationBruteForce : GasStation {
    override fun canCompleteCircuit(gas: IntArray, cost: IntArray): Int {
        val n: Int = gas.size
        var totalSurplus = 0
        var surplus = 0
        var start = 0

        for (i in 0 until n) {
            totalSurplus += gas[i] - cost[i]
            surplus += gas[i] - cost[i]
            if (surplus < 0) {
                surplus = 0
                start = i + 1
            }
        }
        return if (totalSurplus < 0) -1 else start
    }
}
