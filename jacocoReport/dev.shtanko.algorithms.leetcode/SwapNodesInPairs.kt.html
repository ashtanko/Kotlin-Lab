package dev.shtanko.algorithms.leetcode

/**
 * 24. Swap Nodes in Pairs
 * @see <a href="https://leetcode.com/problems/swap-nodes-in-pairs">leetcode page</a>
 */
fun interface SwapNodesInPairs {
    fun swapPairs(head: ListNode?): ListNode?
}

class SwapNodesInPairsSimple : SwapNodesInPairs {
    override fun swapPairs(head: ListNode?): ListNode? {
        if (head?.next == null) {
            return head
        }
        val ans = ListNode.empty()
        ans.next = head
        var curr: ListNode? = ans
        while (curr?.next != null && curr.next?.next != null) {
            val t1 = curr.next
            val t2 = curr.next?.next
            curr.next = t2
            t1?.next = t2?.next
            t2?.next = t1
            curr = curr.next?.next
        }
        return ans.next
    }
}
