package dev.shtanko.di.multibindings

import dagger.Component
import dagger.Module
import dagger.Provides
import dagger.multibindings.ElementsIntoSet
import dagger.multibindings.IntoSet

private const val ONE_STRING = "ABC"
private const val DEF = "DEF"
private const val GHI = "GHI"

/**
 * Multibindings
 * @see <a href="https://dagger.dev/dev-guide/multibindings">leetcode page</a>
 */
@Component(modules = [ModuleA::class, ModuleB::class])
interface MComponent {
    fun strings(): Set<String>
}

@Module
class ModuleA {
    @Provides
    @IntoSet
    fun provideOneString(): String {
        return ONE_STRING
    }
}

@Module
class ModuleB {
    @Provides
    @ElementsIntoSet
    fun provideSomeStrings(): Set<String> {
        return setOf(DEF, GHI)
    }
}
