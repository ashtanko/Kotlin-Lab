package dev.shtanko.algorithms.templates

import dev.shtanko.algorithms.leetcode.ListNode

fun reversingLinkedList(head: ListNode?): ListNode? {
    var curr: ListNode? = head
    var prev: ListNode? = null
    while (curr != null) {
        val nextNode: ListNode? = curr.next
        curr.next = prev
        prev = curr
        curr = nextNode
    }
    return prev
}
