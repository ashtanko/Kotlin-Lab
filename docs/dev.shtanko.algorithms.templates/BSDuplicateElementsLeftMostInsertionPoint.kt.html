package dev.shtanko.algorithms.templates

/**
 * Binary search: duplicate elements, left-most insertion point
 */
private fun fn(arr: IntArray, target: Int): Int {
    var left = 0
    var right = arr.size
    while (left < right) {
        val mid = left + (right - left) / 2
        if (arr[mid] >= target) {
            right = mid
        } else {
            left = mid + 1
        }
    }

    return left
}
