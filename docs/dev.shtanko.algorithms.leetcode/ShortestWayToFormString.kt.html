package dev.shtanko.algorithms.leetcode

object ShortestWayToFormString {
    fun twoPointers(source: String, target: String): Int {
        var t = 0
        var ans = 0
        while (t < target.length) {
            val pt = t
            for (s in source.indices) {
                if (t < target.length && source[s] == target[t]) {
                    t++
                }
            }

            if (t == pt) {
                return -1
            }
            ans++
        }
        return ans
    }
}
