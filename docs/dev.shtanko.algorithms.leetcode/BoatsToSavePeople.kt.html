package dev.shtanko.algorithms.leetcode

/**
 * 881. Boats to Save People
 * @see <a href="https://leetcode.com/problems/boats-to-save-people/">leetcode page</a>
 */
fun interface BoatsToSavePeople {
    fun numRescueBoats(people: IntArray, limit: Int): Int
}

class BoatsToSavePeopleBF : BoatsToSavePeople {
    override fun numRescueBoats(people: IntArray, limit: Int): Int {
        var boatCount = 0
        people.sort()

        var left = 0
        var right: Int = people.size - 1

        while (left <= right) {
            val sum = people[left] + people[right]
            if (sum <= limit) {
                boatCount++
                left++
                right--
            } else {
                boatCount++
                right--
            }
        }
        return boatCount
    }
}
