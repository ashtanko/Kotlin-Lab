package dev.shtanko.algorithms.hackerrank

import dev.shtanko.algorithms.leetcode.MOD

object CuttingBoards {

    fun perform(x: IntArray, y: IntArray): Long {
        var horizontalCuts = 0
        var verticalCuts = 0
        val m = x.map { true to it }.asSequence() + y.map { false to it }.asSequence()
        return m.sortedByDescending { it.second }
            .map { (isHorizontalCut, cost) ->
                if (isHorizontalCut) {
                    horizontalCuts++
                    cost * (verticalCuts + 1)
                } else {
                    verticalCuts++
                    cost * (horizontalCuts + 1)
                }
            }.sum() % MOD.toLong()
    }
}
