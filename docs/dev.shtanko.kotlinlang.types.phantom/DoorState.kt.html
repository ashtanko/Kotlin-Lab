package dev.shtanko.kotlinlang.types.phantom

sealed class DoorState

object Open : DoorState()
object Closed : DoorState()

class DoorViolation(val state: DoorState) {
    fun open() = DoorViolation(Open)
    fun close() = DoorViolation(Closed)
}

// solution
class Door<out T : DoorState>(val state: T)

fun Door<Closed>.open() = Door(Open)
fun Door<Open>.close() = Door(Closed)
