package dev.shtanko.di.manual

private const val ESPRESSO_TMP = 3L // 30 * 1000
private const val AMERICANO_TMP = 3L // 30 * 1000
private const val LATTE_TMP = 3L // 75 * 1000

internal interface Coffee {
    fun brewingTime(): Long
}

internal class Espresso : Coffee {
    override fun brewingTime(): Long {
        return ESPRESSO_TMP
    }
}

internal class Americano : Coffee {
    override fun brewingTime(): Long {
        return AMERICANO_TMP
    }
}

internal class Latte : Coffee {
    override fun brewingTime(): Long {
        return LATTE_TMP
    }
}
